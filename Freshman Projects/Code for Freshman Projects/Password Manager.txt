playSound("Ultra-Instinct-Opening-Theme---Sound-Effect-(HD).mp3", false);

// Declare variables 
// Main variable columns
var passwords = getColumn("Passwords", "password");
var passwordCategories = getColumn("Passwords", "category");
var passwordStrengths = getColumn("Passwords", "strength");

// Other variables
var strengthenedPassword = "";
var passwordStrengthNumber = 0;
var passwordStrengthLevel = "A little bit";
var selectedCategory = "password-related";
var randomPasswordIndex = 0;
var characterList = [1,2,3,4,5,6,7,8,9,0,"q","w","e","r","t","y","u","i","o","p","a","s","d","f","g","h","j","k","l","z","x","c","v","b","n","m", "!", "@", "#", "$","%", "^", "&","*","(", ")", "Q","W","E","R","T","Y","U","I","O","P","A","S","D","F","G","H","J","K","L","Z","X","C","V","B","N","M"];

// Screen count
var screenNumber = 1;

// Filtered Lists
var filteredPasswords = [];
var filteredStrengthenedPassword = [];

// On events
//Next and back buttons
onEvent("beginButton", "click", function( ) {
  screenNumber++;
  changeScreen();
});
onEvent("filteringToolBackButton", "click", function( ) {
  screenNumber--;
  changeScreen();
});
onEvent("filteringToolNextButton", "click", function( ) {
  screenNumber++;
  changeScreen();
});
onEvent("randomGeneratorBackButton", "click", function( ) {
  screenNumber--;
  changeScreen();
});
onEvent("randomGeneratorNextButton", "click", function( ) {
  screenNumber++;
  changeScreen();
});
onEvent("strengthingToolBackButton", "click", function( ) {
  screenNumber--;
  changeScreen();
});

//Filtering tool screen 
onEvent("categoryDropdown", "change", function( ) {
  selectedCategory = getText("categoryDropdown");
  updateFilteringTool();
});
onEvent("strengthNumberInput", "keyup", function( ) {
  passwordStrengthNumber = getNumber("strengthNumberInput");
  updateFilteringTool();
});

// Random password generator screen
onEvent("randomGeneratorButton", "click", function( ) {
  randomPasswordIndex = randomNumber(1, passwords.length);
  setText("randomGeneratorOutput", "Password: " + passwords[randomPasswordIndex] + "\n" + "Password strength: " + passwordStrengths[randomPasswordIndex] + "\n" + "Password type: " + passwordCategories[randomPasswordIndex] + "\n" + randomPasswordIndex + "th most common password");
});


// Password strengthener screen
onEvent("passwordInput", "keyup", function( ) {
  strengthenedPassword = getText("passwordInput");
  updateStrengtheningTool();
});
onEvent("strengthingPowerInput", "change", function( ) {
  passwordStrengthLevel = getText("strengthingPowerInput");
  updateStrengtheningTool();
});

// Functions

//Purpose - to allow the user to navigate the app
//It uses conditions to check if the screenNumber variable matches with its corresponding screen
//if it does, the screen is changed
function changeScreen() {
  if (screenNumber == 1) {
    setScreen("appIntroduction");
  } else if (screenNumber == 2) {
    setScreen("filteringTool");
  } else if (screenNumber == 3) {
    setScreen("randomPasswordGenerator");
  } else {
    setScreen("strengthingTool");
  }
}

//Purpose - to filter the user’s passwords based on their selected category and minimum strength
//It tranverses the passwords list and filters it based on if the category and number matches up with the user’s selected one
//Then it sets the text of the output box with that list
function updateFilteringTool() {
  filteredPasswords = [];
  for (var i = 0; i < passwords.length; i++) {
    if (selectedCategory == passwordCategories[i]) {
      if (passwordStrengthNumber <= passwordStrengths[i]) {
        appendItem(filteredPasswords, passwords[i]);
    }
  }
}
  setText("filteringToolOutput", "Here are your passwords: " + filteredPasswords);
}

//Purpose - update the screen of the strengthening tool
//Works by looping through each character of the user’s input
//It adds each character and then one or multiple random characters everytime one is added
//It then joins the list and displays it on screen
function updateStrengtheningTool() {
  filteredStrengthenedPassword = [];
  for (var i = 0; i < strengthenedPassword.length; i++) {
    appendItem(filteredStrengthenedPassword, strengthenedPassword[i]);
    appendItem(filteredStrengthenedPassword, characterList[randomNumber(0, characterList.length-1)]);
    if (passwordStrengthLevel == "A lot") {
      appendItem(filteredStrengthenedPassword, characterList[randomNumber(0, characterList.length-1)]);
    } else if ((passwordStrengthLevel == "MAXIMUM STRENGTH")) {
      for (var j = 0; j < 10; j++) {
        appendItem(filteredStrengthenedPassword, characterList[randomNumber(0, characterList.length-1)]);
      }
    }
  }
  setText("strengthenedPasswordOutput", filteredStrengthenedPassword.join(""));
}
