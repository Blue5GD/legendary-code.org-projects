playSound("Ultra-Instinct-Opening-Theme---Sound-Effect-(HD).mp3", false);

//Buttons to navagate the quiz
onEvent("introductionButton", "click", function( ) {
  setScreen("quizScreen");
});
onEvent("backButton", "click", function( ) {
  setScreen("introductionScreen");
});

//Declare variables
var rating = 0;
var memory = "good";
var aggression = "true";
var gameplan = "by finding tactics";
var username = "*no name inputted*";
var recommendation = "";

//On event blocks
onEvent("ratingDropdown", "change", function( ) {
  rating = getNumber("ratingDropdown");
  updateScreen();
  playSound("sound://category_animals/pig.mp3", false);
});
onEvent("memoryDropdown", "change", function( ) {
  memory = getText("memoryDropdown");
  updateScreen();
  playSound("AUGHHHH-sound-effect-tiktok-snoring-meme.mp3", false);
});
onEvent("aggressionDropdown", "change", function( ) {
  aggression = getText("aggressionDropdown");
  updateScreen();
  playSound("Samsung-notification-sound-effect-(-no-copyright).mp3", false);
});
onEvent("gameplanDropdown", "change", function( ) {
  gameplan = getText("gameplanDropdown");
  updateScreen();
  playSound("Guy-Yelling-Among-Us-Sound-Effect.mp3", false);
});
onEvent("nameInput", "keyup", function( ) {
  username = getText("nameInput");
  updateScreen();
});


// Purpose - This function's purpose is to update the recommendation of the program depending on what the user selects in the dropdown boxes. 
// Functionality: First, the function uses several if statements to check which opening to recommend.  
//It then changes the text of the output box using the determined opening recommendations, 
//and displays the other information inputted by the user in a clean format along with some other text. 

function updateScreen() {
   
   if (rating > 500 && (memory != "bad") &&(aggression == "false") && (gameplan == "by finding tactics")) {
     recommendation = "Solid openings (EX: The Ruy Lopez and Two Knights Defense)";
   } else if (rating > 500 && (memory != "bad") &&(aggression == "false") && (gameplan == "by outmaneuvering my opponent")) {
     recommendation = "Defensive openings (EX: Caro Kann, French Defense, QGD)";
   } else if (rating > 500 && (memory != "bad") &&(aggression == "true")) {
     recommendation = "Tricky, sharp, and complicated openings (EX: The Sicilian, Evan’s Gambit, and Grunfeld)";
   } else if (rating > 500 && (memory == "bad") && (aggression == "true")) {
     recommendation = "Simpler but still tricky openings (EX: Vienna and Fried Liver Attack)";
   } else if (rating > 500 && (memory == "bad") && (aggression == "false")) {
     recommendation = "d4 Nf3 c4 systems without much theory";
   } else if (rating <= 500 && (memory != "bad") && (aggression == "false")) {
     recommendation = "Simple setup based openings (EX: Slav Defense, London System)";
   } else if (rating <= 500 && (memory != "bad") && (aggression == "true")) {
     recommendation = "Simple, open, and straight forward openings (EX: 4 Knights Scotch, Russian game)";
   } else {
     recommendation = "None! Maybe you should switch to checkers.";
   }
   
   setText("quizOutput", "What's up " + username + ". Because your rating is close to " + rating + ", your memory is " + memory + ", you prefer to win games " + gameplan + ", and it is " + aggression + " that you are aggressive, I would recommend:" + "\n" + "\n" + recommendation);
}
