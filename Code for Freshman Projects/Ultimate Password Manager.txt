// Useful Functions library 1

//Purpose - to allow the user to navigate the app
//It uses conditions to check if the screenNumber variable matches with its corresponding screen
//if it does, the screen is changed
function changeScreen() {
  if (screenNumber == 1) {
    setScreen("introScreen");
  } else if (screenNumber == 2) {
    setScreen("filteringTool");
  } else if (screenNumber == 3) {
    setScreen("randomPasswordsScreen");
  } else if ((screenNumber == 4)) {
    setScreen("generatorTool");
  } else if ((screenNumber == 5)) {
    setScreen("strengtheningTool");
  } else if ((screenNumber == 6)) {
    setScreen("publicNYCwifiFinder");
  } else {
    setScreen("randomNYCPublicWifiScreen");
  }
}
//Gets an item from the specified index
//Can be used with the random number function to get random passwords/public wifi locatiosn easier
function randomItem(randomIndex, dataColumn) {
  var theRandomItem = dataColumn[randomIndex];
  return theRandomItem;
}


//Password manager Library

//Allows the user to get a list of filtered passwords
//based on the category and strength of the password they input
//Strengh - the level of strength the password has with 0 being the lowest and 48 being the highest
//Category - check the passwords data column to see the list of different categories
function getFilteredPasswords(strength, category, length) {
  var filteredPasswordList = [];
  for (var i = 0; i < passwords.length; i++) {
    if (category == passwordCategories[i]) {
      if (strength <= (passwordStrengths[i])) {
        if (length <= ((passwords[i]).length)) {
          appendItem(filteredPasswordList, passwords[i]);
        }
      }
    }
  }
  return filteredPasswordList;
}
//Generates a completely random password from the user’s given list
//Generates as many characters as the user specifies
//Length - the number of characters in the user's generated password
//Character list - the possible characters that will be generated. 
//So if you gave the character list "ABC", the only possible characters that will be generated will be A B and C
function generatePassword(length, characterList) {
  var generatedPassword = [];
  for (var i = 0; i < length; i++) {
    appendItem(generatedPassword, characterList[randomNumber(0, characterList.length-1)]);
  }
  return (generatedPassword.join(""));
}
//Strengthens the user’s given password
//The level parameter is the degree of strength, or length the password is strengthened
//Fills the user’s password with random characters from their specified list
//Level - basically the amount of characters added between each character in the password
function strengthenPassword(password, level, characterList) {
  var newPassword = [];
  for (var i = 0; i < password.length; i++) {
    appendItem(newPassword, password[i]);
    for (var j = 0; j < level; j++) {
      appendItem(newPassword, characterList[randomNumber(0, characterList.length-1)]);
    }
  }
  return (newPassword.join(""));
}


//Functions that update specific screens
//Updates the strengthening tool by using if statements and the strengthenPassword function
function updateStrengtheningTool() {
  if (selectedList == "All characters") {
    setText("strengtheningToolOutput", "Here is your strengthened password: " + strengthenPassword (passwordToStrengthen, passwordStrengthLevel, allCharacters));
  } else {
    setText("strengtheningToolOutput", "Here is your strengthened password: " + strengthenPassword (passwordToStrengthen, passwordStrengthLevel, numbersAndLetters));
  }
}
function updateFilteringTool() {
  setText("filteringToolOutput", "Here are your passwords: " + getFilteredPasswords(filteringStrength, filteringCategory, filteringLength));
}
function updateWifiFinder() {
  setText("wifiFinderOutput", "Here's a list of public Wifi locations based on your selected options: " + filterPublicWifi(selectedProvider, selectedBorough, selectedAccessLocation) + "\n" + "\n" + "If it doesn't work, it's because your combination of selected boroughs, providers, and access locations doesn't exist. Check the data column to see ones that exist, or keep trying different combinations.");
}


// New York City wifi function
// Transverses through the list to generate a list of wifi locations
function filterPublicWifi(provider, borough, accessLocation) {
  var filteredWifiList = [];
  for (var i = 0; i < wifiLocations.length; i++) {
    if (provider == wifiProviders[i]) {
      if (accessLocation == (wifiAccessLocations[i])) {
        if (borough == (wifiBoroughs[i])) {
          appendItem(filteredWifiList, wifiLocations[i]);
        }
      }
    }
  }
  return filteredWifiList;
}
